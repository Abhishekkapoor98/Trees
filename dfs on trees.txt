void add_edge(vi graph[],int u, int v)
{
	graph[u].PB(v);
	graph[v].PB(u);
}

void dfs(int root,vi graph[], int parent)
{
	cout<<root<<" ";

	for(auto itr : graph[root])
		if(itr != parent)
			dfs(itr,graph,root);
}

}
